def add_job(job_id)
  # Notice that the `more_attributes` serialization mechanism converts the set into an array
  ma['jobs'].present? ? ma['jobs'] = Set.new(ma['jobs']).add(job_id) : ma['jobs'] = Set.new([job_id])
  self.save!
end
